# custom elements

    @@@ php

    class :ui:box extends :x:element {
      protected function render() {
        return
          <div class="box">
            <div class="inner">
              {$this->getChildren()}
            </div>
          </div>
      }
    }

    echo 
      <ui:box>Hello World</ui:box>


    // <div class="box">
    //   <div class="inner">
    //    Hello World
    //   </div>
    // </div>

# custom attributes

    @@@ php

    class :ui:page extends :x:element {
      attribute
        string title="No Title",
        enum { "post", "page" } type

      protected function render() {
        return
          <div class={$this->getAttribute("type")}>
            <h1>{$this->getAttribute("title")}</h1>
            <div class="body">
            </div>
          </div>
        }
    }

    echo
      <ui:page title="10 things" type="post">
        Lorem Ipsum
      </ui:page>

    // <div class="post">
    //   <h1>10 things</h1>
    //   <div class="body">
    //     Lorem Ipsum
    //   </div>
    // </div>
